\begin{frame}
  \frametitle{Pointers}
  \code[language=c++14,width=.4\linewidth]{equiv.cpp}
  \begin{itemize}
    \item \texttt{*p} signifies ``value pointed to by \texttt{p}''
    \item \texttt{*p} is just another way to mean \texttt{x}
    \item E.g.~\texttt{x++} is equivalent to \texttt{(*p)++}
  \end{itemize}
\end{frame}
