\begin{frame}
  \frametitle{Access Modifiers}
  \begin{itemize}
    \item \texttt{public}
          \begin{itemize}
            \item Accessible to everyone
          \end{itemize}
    \item \texttt{protected}
          \begin{itemize}
            \item Accessible to subclasses
            \item Different from Java!
          \end{itemize}
    \item \texttt{private}
          \begin{itemize}
            \item Accessible within class
          \end{itemize}
  \end{itemize}
\end{frame}
